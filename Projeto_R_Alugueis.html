Parte 1: Web Scraping da Base de Dados

O código em R desenvolvido realiza web scraping do site dfimoveis.com.br para extrair informações sobre apartamentos disponíveis para aluguel no Distrito Federal. Ele utiliza a função lapply para aplicar a função remove_non_numeric a múltiplas colunas do data frame, removendo caracteres não numéricos e convertendo os valores para formato numérico onde aplicável, como preço, área em metros quadrados, número de quartos, suítes, vagas de garagem e CRECI.

Além disso, o script faz uso de pacotes como rvest para manipulação de HTML, stringr para operações com strings e dplyr para transformações e filtragens de dados. Após a extração e limpeza dos dados, os resultados são consolidados em um data frame único e exportados para um arquivo CSV chamado apartamentos.csv. O código finaliza mostrando o número total de apartamentos encontrados e um resumo estatístico dos dados coletados, facilitando análises posteriores do mercado imobiliário na região.

# URL base e número da página inicial
url_base <- 'https://www.dfimoveis.com.br/aluguel/df/todos/apartamento'
pagina_atual <- 1
total_apartamentos <- 0

# Bairros a serem procurados
bairros <- c(
  "AGUAS CLARAS", "TAGUATINGA", "GUARA", "SAMAMBAIA",
  "NUCLEO BANDEIRANTE", "RIACHO FUNDO", "SOBRADINHO", "CEILANDIA", "VICENTE PIRES",
  "CRUZEIRO", "GAMA", "PLANALTINA", "SANTA MARIA", "JARDIM BOTANICO",
  "PARANOA", "CANDANGOLANDIA", "RECANTO DAS EMAS", "SAO SEBASTIAO", "SETOR INDUSTRIAL",
  "VARJAO", "ALPHAVILLE", "AGUAS LINDAS DE GOIAS", "FORMOSA", "LUZIANIA",
  "ASA NORTE", "ASA SUL", "NOROESTE", "SUDOESTE", "PARK SUL",
  "LAGO NORTE", "JARDINS MANGUEIRAL", "JARDIM BOTANICO", "OCTOGONAL", "LAGO SUL",
  "GRANJA DO TORTO", "PARK WAY", "SETOR MILITAR URBANO", "SIG", "VALPARAISO DE GOIAS"
)

# Lista para armazenar os dados dos apartamentos
apartamentos_data <- list()

while(TRUE) {
  # Faz a solicitação HTTP para a página atual
  response <- GET(paste0(url_base, "?pagina=", pagina_atual))
  
  # Verifica se a solicitação foi bem-sucedida
  if(status_code(response) != 200) {
    print(paste("Erro ao acessar a página", pagina_atual, ". Status code:", status_code(response)))
    break
  }
  
  site <- read_html(content(response, as = "text"))
  
  # Encontra todos os elementos que representam os apartamentos na página
  apartamentos <- site %>% html_nodes('a.new-card')
  
  # Se não houver mais apartamentos, sai do loop
  if(length(apartamentos) == 0) {
    break
  }
  
  # Loop através dos apartamentos na página atual
  for(apartamento in apartamentos) {
    # Extrai os dados relevantes de cada apartamento
    endereco_texto <- apartamento %>% html_node('h2.new-title.phrase') %>% html_text(trim = TRUE)
    preco_texto <- apartamento %>% html_node('div.new-price') %>% html_nodes('h4') %>% html_text(trim = TRUE)
    detalhes_elem <- apartamento %>% html_node('ul.new-details-ul')
    creci <- apartamento %>% html_node('div.creci p') %>% html_text(trim = TRUE)
    
    # Procura por cada bairro na string do endereço
    bairro <- NA
    for(b in bairros) {
      if(str_detect(endereco_texto, b)) {
        bairro <- b
        break
      }
    }
    
    preco <- if(length(preco_texto) > 0) preco_texto[1] else NA
    valor_metro_quadrado <- if(length(preco_texto) > 1) preco_texto[2] else NA
    
    metros_quadrados <- quartos <- suites <- vagas <- NA
    
    # Verifica se os detalhes do apartamento foram encontrados antes de extrair o texto
    if(!is.null(detalhes_elem)) {
      metros_quadrados <- detalhes_elem %>% html_node('li.m-area') %>% html_text(trim = TRUE)
      detalhes <- detalhes_elem %>% html_nodes('li')
      for(item in detalhes) {
        texto <- item %>% html_node('span') %>% html_text(trim = TRUE)
        if(str_detect(texto, regex('quarto', ignore_case = TRUE))) {
          quartos <- texto
        } else if(str_detect(texto, regex('suíte', ignore_case = TRUE))) {
          suites <- texto
        } else if(str_detect(texto, regex('vaga', ignore_case = TRUE))) {
          vagas <- texto
        }
      }
    }
    
    # Adiciona os dados do apartamento à lista de apartamentos
    apartamentos_data <- append(apartamentos_data, list(data.frame(
      Endereco = endereco_texto,
      Bairro = bairro,
      Preco = preco,
      Valor_m2 = valor_metro_quadrado,
      Metros_Quadrados = metros_quadrados,
      Quartos = quartos,
      Suites = suites,
      Vagas = vagas,
      CRECI = creci,
      stringsAsFactors = FALSE
    )))
    
    # Incrementa o contador de apartamentos
    total_apartamentos <- total_apartamentos + 1
  }
  
  # Incrementa o número da página atual
  pagina_atual <- pagina_atual + 1
}

# Converte a lista de data frames em um único data frame
df <- bind_rows(apartamentos_data)

# Função para remover caracteres não numéricos
remove_non_numeric <- function(text) {
  return(gsub("\\D", "", text))
}

# Colunas a serem limpas
cols_to_clean <- c("Preco", "Valor_m2", "Metros_Quadrados", "Quartos", "Suites", "Vagas", "CRECI")

# Aplicar a função para remover caracteres não numéricos e converter para numérico
df[cols_to_clean] <- lapply(df[cols_to_clean], function(col) as.numeric(remove_non_numeric(col)))

# Converter o dataframe resultante em um data frame regular
df <- as.data.frame(df)

# Salva o DataFrame em um arquivo CSV
write.csv(df, "apartamentos.csv", row.names = FALSE)

# Imprime o total de apartamentos encontrados
print(paste("Total de apartamentos encontrados:", total_apartamentos))

Parte 2: Manipulação, Limpeza e Transformação de Dados

Esse trecho de código faz a limpeza dos outliers e tambem substitui todos os valores faltantes (NA) por 0. Utilizamos a função filter para remover registros com valores extremos em várias colunas (Vagas, Metros_Quadrados, Preco, Valor_m2) Em seguida, a função mutate_all substitui todos os valores NA em todas as colunas por 0. Finalmente, summary(df_limpo) é usado para verificar a distribuição dos dados após a limpeza.

# Remover outliers e substituir NA por 0
df_limpo <- df %>%
  filter(Vagas <= 10) %>%                  # Coluna Vagas: remover valores acima de 10
  filter(Metros_Quadrados >= 3) %>%        # Coluna Metros_Quadrados: remover valores abaixo de 3m²
  filter(Preco <= 80000) %>%               # Coluna Preco: remover valores acima de 80.000,00
  filter(Valor_m2 > 2) %>%                 # Coluna Valor_m2: remover valores menores ou iguais a 2
  mutate_all(~replace(., is.na(.), 0))     # Substituir NA por 0 em todas as colunas

# Verificar a distribuição dos dados após a limpeza

Contar o número de apartamentos por bairro em todos os bairros

Esse trecho agrupa os dados pelo valor da coluna Bairro e conta o número de registros (apartamentos) em cada bairro. O resultado é armazenado em contagem_por_bairro. Ordena os bairros pela contagem de apartamentos em ordem decrescente para facilitar a visualização. Cria um gráfico de barras para visualizar o número de apartamentos por bairro. Adciona rótulos informativos e formatação do gráfico para facilitar a leitura.

# Contar o número de apartamentos por bairro em todos os bairros
contagem_por_bairro <- df_limpo %>%
  group_by(Bairro) %>%
  summarise(Numero_de_Apartamentos = n())

# Ordenar os bairros pela contagem de apartamentos
contagem_por_bairro <- contagem_por_bairro[order(contagem_por_bairro$Numero_de_Apartamentos, decreasing = TRUE), ]

# Plotar o gráfico de barras com rótulos de quantidade em cada barra
ggplot(contagem_por_bairro, aes(x = reorder(Bairro, Numero_de_Apartamentos), y = Numero_de_Apartamentos)) +
  geom_bar(stat = "identity", fill = "purple") +
  geom_text(aes(label = Numero_de_Apartamentos), vjust = -0.5, size = 3) +  # Adicionar rótulos de quantidade
  labs(title = "Número de Apartamentos para Alugar por Bairro", x = "Bairro", y = "Número de Apartamentos") +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

Média de Preço por Bairro

Calcula a média de preço por bairro usando aggregate do base R. Ordena os resultados pela média de preço em ordem decrescente. Imprime a tabela resultante e cria um gráfico de barras (ggplot) que mostra a média de preço por bairro.

# Média de preço por bairro
media_preco_por_bairro <- aggregate(Preco ~ Bairro, data = df_limpo, FUN = mean)
media_preco_por_bairro <- media_preco_por_bairro[order(media_preco_por_bairro$Preco, decreasing = TRUE), ]

# Gráfico de barras da média de preço por bairro
ggplot(media_preco_por_bairro, aes(x = reorder(Bairro, -Preco), y = Preco)) +

Análise de Relação entre Preço e Tamanho (metros quadrados)

Cria um scatterplot (ggplot) que mostra a relação entre o tamanho (metros quadrados) e o preço dos apartamentos. Usa geom_point para os pontos de dados e geom_smooth para traçar uma linha de tendência linear (method = “lm”). Define rótulos para os eixos (labs) e ajusta as escalas dos eixos (scale_x_continuous e scale_y_continuous).

O gráfico de dispersão com linha de tendência linear apresenta a relação entre o preço dos apartamentos e seu tamanho em metros quadrados. A linha de tendência vermelha indica uma relação positiva, sugerindo que apartamentos maiores tendem a ter preços mais altos. A dispersão dos pontos ao redor da linha mostra a variabilidade nos preços para apartamentos de tamanhos semelhantes, destacando outliers que podem indicar casos com preços atípicos. Esse gráfico é útil para visualizar padrões gerais na relação entre preço e tamanho, bem como para entender a adequação de um modelo linear simples para descrever essa relação no mercado imobiliário.

# Scatterplot de preço por tamanho (metros quadrados)
ggplot(df_limpo, aes(x = Metros_Quadrados, y = Preco)) +
  geom_point(color = "darkorange") +
  geom_smooth(method = "lm", se = FALSE, color = "purple") +
  labs(title = "Preço em Função do Tamanho",
       x = "Tamanho (m²)", y = "Preço (R$)") +
  scale_x_continuous(breaks = seq(0, ceiling(max(df_limpo$Metros_Quadrados) / 50) * 50, by = 50)) +
  scale_y_continuous(breaks = seq(0, ceiling(max(df_limpo$Preco) / 5000) * 5000, by = 5000))
